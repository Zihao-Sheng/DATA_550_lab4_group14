import numpy as np, pandas as pd, altair as alt
import statsmodels.api as sm

x, y, g = "information_density", "distinct_syllables", "continent"
df = languages.query("iso_lang != 'FRA'")[["language", x, y, g]].dropna()

pred = []
for k, d in df.groupby(g):
    X = sm.add_constant(d[x])
    m = sm.OLS(d[y], X).fit()
    grid = np.linspace(d[x].min(), d[x].max(), 80)
    pr = m.get_prediction(sm.add_constant(grid)).summary_frame(alpha=0.05)
    pred.append(pd.DataFrame({x:grid, g:k, "yhat":pr["mean"], "lo":pr["mean_ci_lower"], "hi":pr["mean_ci_upper"]}))
pred = pd.concat(pred, ignore_index=True)

points = alt.Chart(df).mark_point(size=60, opacity=0.6).encode(
    x=alt.X(f"{x}:Q", scale=alt.Scale(zero=False)),
    y=alt.Y(f"{y}:Q", scale=alt.Scale(zero=False)),
    color=f"{g}:N",
    tooltip=[f"{g}:N", "language:N", f"{x}:Q", f"{y}:Q"]
)

band = alt.Chart(pred).mark_area(opacity=0.25).encode(
    x=f"{x}:Q", y="lo:Q", y2="hi:Q", color=f"{g}:N"
)

line = alt.Chart(pred).mark_line().encode(
    x=f"{x}:Q", y="yhat:Q", color=f"{g}:N"
)

plot_A=points + band + line

    obs
    .groupby(["iso_lang", "continent"], as_index=False)

Plot A shows that languages with higher information density tend to have a larger inventory of distinct syllables, suggesting that increased phonological complexity is one way languages encode more information. However, this alone does not describe how efficiently information is conveyed over time. Plot B therefore examines speech rate and reveals a clear trade-off: as information density increases, speech rate generally decreases, and no language is simultaneously high in both. Together, these results indicate that languages achieve comparable overall rates of information transmission by balancing information density and speech rate in different ways, consistent with a shared limit on human auditory processing.
